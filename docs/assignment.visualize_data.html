3.1 Group: normal

(def norm-dat
  (-> data
      (ds/select-rows #(= (:group %) "normal"))))
^kind/vega
(let [dat (ds/rows norm-dat :as-maps)
      column-names cols-of-interest]
  {:data   {:values dat}
   :repeat {:column column-names}
   :spec   {:mark     "bar"
            :encoding {:x     {:field {:repeat "column"}
                               :bin   {:steps [1 3]} :type "quantitative"}
                       :y     {:aggregate "count"}
                       :color {:field :group}}}})
